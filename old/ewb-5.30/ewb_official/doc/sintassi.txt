ifblock= if calcexp { [codice] }
        [elseif calcexp { [codice] } ] { else {[codice]} } ;
forblock= for (calcexp; calcexp; calcexp) { [codice] };

whileblock= while calcexp { [codice] };
downwhile= do { [codice] } while calcexp;

fundecl= sub name([varname [,varname] ])
          [ { [codice] } ];


calcexp=exp  cfr calcexp | exp
cfr = \= | == | != | <= | >= | > | < | eq | gt | ge | lt | le
'nota: se uno dei due el. e' una stringa e' un conf. di stringhe.
exp=e . exp | e
e = t+e | t-e | t
t = f*t | f/t | f%t | f
f= b\|f | b\|\|f | b
b= c & b | c && b | c
c= ! calcexp | numvalue | strvalue | tablename | identifier | (calcexp)
identifier = variable | funzione

'cambiare qui per usare le liste. 


funzione = predef | fname calcexp [,calcexp]

numvalue = [0..9]. [0..9]
strvalue = \" carattere \" | \[carattere\]
variable =a..z|A..Z [a..z | A..Z | 1..9 | _ ]
fname =variabile
