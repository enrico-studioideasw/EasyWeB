sub programma($)
{ my $code=shift;
  while ($code ne "")
  { $code=codice($code);
  }
}

sub codice($)
{ my $code=shift;
  $code=delperc($code);
  if (is_commento($code))
  { $code=commento($code);
  }
  elsif ( (substr($code, 0, 5) eq "table") && isperc(substr($code, 5)) )
  { $code=tabdecl($code);
  }
  elsif ( (substr($code, 0, 5) eq "var") && isperc(substr($code, 3)) )
  { $code=vardecl($code);
  }
  elsif ( (substr($code, 0, 2) eq "if") && isperc(substr($code, 2)) )
  { $code=ifblock($code);
  }
  elsif ( (substr($code, 0, 5) eq "while") && isperc(substr($code, 5)) )
  { $code=whileblock($code);
  }
  elsif ( (substr($code, 0, 2) eq "do") && isperc(substr($code, 2)) )
  { $code=downwhile($code);
  }
  elsif ( (substr($code, 0, 3) eq "sub") && isperc(substr($code, 3)) )
  { $code=fundecl($code);
  }
  else
  { if ($code ne "") { $code=calcexp($code); };
  }
}

#delperc:elimina i caratteri di spaziatura, conta le righe e le riporta su stdout per debug.
sub isperc($)
{ my $code=shift;
  my $a;
  $a=substr($code, 0, 1);
  return (($a eq ' ')|| ($a eq "\t") || ($a eq "\n"));
}
sub delperc($)
{ my $code=shift;
  my $a;
  do
  { $a=substr($code, 0, 1);
    if (($a eq ' ') || ($a eq '\t') || ($a eq '\n'))
    { $code=substr($code, 1, strlen($code));
      if ($a eq "\n") { $curr_row=(split("\n", $code))[0];
                        $num_row++;
			print "##\t".$curr_row."\n";
		      };
      $a = "jump";
    }
  } while ($a ne "jump");
  return $code;
}

sub errore($)

sub is_commento($) {my $code=shift;
  return ((substr $code, 0, 1) eq '\'');};

sub commento($)
{ my $code=shift;
  while (substr($code, 0,1) ne "\n")
  { $code=substr($code, 1, strlen($code));
  }
  return delperc($code);
}

sub calcexp($)
{ my $code=shift;

  $code=delperc($code);
  $code=exp($code);
  $code=delperc($code);
  while (is_cfr($code))
  { $code=cfr($code);
    $code=delperc($code);
    $code=exp($code);
  }
  return $code;
}

sub is_cfr($)
{ my $code=shift;
  my $a=substr($code, 0, 1);
  my $b=substr($code, 0, 2);
  return ( ($a eq "=") || ($a eq "<") || ($a eq ">") || 
           ($b eq "==")|| ($b eq "<=")|| ($b eq ">=")||
           ($b eq "ge")|| ($b eq "le")|| ($b eq "lt")||
           ($b eq "eq"));
}

sub term_c($)
{ my $code=shift;
  $code=delperc($code);
  if (substr($code, 0, 1) eq "!")
  { print "!";
    $code=delperc($code);
    $code=calcexp($ccode);
  } 
  elsif (substr($code, 0, 1) eq "(")
  { print "(";
    $code=delperc($code);
    $code=calcexp($ccode);
    $code=delperc($code);
    if (substr($code,0, 1) ne ")")
    { errore "manca parentesi chiusa";
    } 
    $code=substr($code, 1);
  }
  elsif(is_identifier($code))
  { $code=identifier($code);
  }  
  elsif(is_numvalue($code))

  }
  elsif(is_stringvalue($code))
  { $code=stringvalue($code);
  }
  else
  { errore "termine sconosciuto.";
  }
  $code=delperc($code);
  return $code;
}

sub tabdecl($)
{ #Qui ho una dichiarazione di tabella.. mi serve solo per l' ambiente.
  my $code=shift;
  $code=substr($code, 6);
  $code=delperc($code);
  my $name;
  my @values=();
  if (substr($code, 0, 1) ne "=") errore "mi aspettavo \"=\". ";
  $code=substr($code,1);
  $code=delperc($code);
  #variable e' un token elementare: non taglia il code e lo devo fare io a mano.
  $name=variable($code);
  $code=substr($code, length($name));
  $code=delperc($code);

  while (substr($code, 0,1) eq ",")
  { $code=substr($code, 1);
    $code=delperc($code);
    $values[$#values+1]=variable($code);
    $code=substr($code, length($values[$#values]);
    $code=delperc($code);
  }
  if (substr($code, 0, 1) ne ";") { errore "manca ; a fine riga."; };
  $tables[$#tables+1]=$name.":".join(":",$values);
  #ogni elemento di una tabella deve diventare una variabile.
  #questo se la variabile non esiste o e' gia' dichiarata da una tabella.
  my $v;
  foreach $v(@values)
  { my $toadd=1;
    foreach $k(@variables)
    { if ($k eq $v.":v")
      $toadd=0;
    }
    if ($toadd)
    { print "my \$".$v."=\"\"; \n";
      $variables[$#variables+1]= $v.":t"; 
    }
  }
  return $code;
}
